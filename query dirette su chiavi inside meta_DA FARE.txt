Sì — se prevedi di fare ricerche/filtri su campi dentro meta (es. email, experienceId) è consigliabile convertire meta a tipo Json nel DB. Di seguito le ragioni brevi, e poi i passi esatti, senza salto.

Perché conviene

Query/WHERE su chiavi interne molto più semplici e performanti (Postgres jsonb/SQLite json1).
Eviti parsing runtime (read-time) e trasformazioni fragili.
Più affidabile per aggregazioni, export e indici parziali.
Contro/attenzioni

Migration da String→Json va fatta con backup e conversione dei dati esistenti.
Alcune funzionalità (full-text) richiedono workarounds a seconda del DB.
Se usi SQLite il supporto JSON è meno potente che in Postgres ma comunque utile.

const { PrismaClient } = require('@prisma/client');
(async () => {
  const prisma = new PrismaClient();
  try {
    console.log('Start normalize meta -> Json');
    const batch = 200;
    let skip = 0;
    while (true) {
      const rows = await prisma.audit.findMany({ orderBy: { id: 'asc' }, take: batch, skip });
      if (!rows.length) break;
      for (const r of rows) {
        if (!r.meta) continue;
        // if meta already JSON type after schema change, skip if prisma returns object
        if (typeof r.meta !== 'string') continue;
        try {
          const obj = JSON.parse(r.meta);
          await prisma.audit.update({ where: { id: r.id }, data: { meta: obj } });
          console.log('Updated id', r.id);
        } catch (e) {
          // not JSON parseable: wrap as raw
          await prisma.audit.update({ where: { id: r.id }, data: { meta: { raw: String(r.meta) } } });
          console.log('Wrapped id', r.id);
        }
      }
      if (rows.length < batch) break;
      skip += batch;
    }
    console.log('Done');
  } catch (e) {
    console.error('Error', e);
  } finally {
    await prisma.$disconnect();
  }
})();
